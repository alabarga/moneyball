create table PLAYERS (
	PID int,
	Name varchar(100) DEFAULT NULL,	
	TID	int DEFAULT NULL,
	Position char(2) DEFAULT NULL,	
	Other_Teams varchar(100) DEFAULT NULL,

	FOREIGN KEY (TID) REFERENCES TEAMS(TID)
